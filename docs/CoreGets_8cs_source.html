38 var nts = new Util.Basic.TimeX.DateTimeX(DateTime.Now).ToMiSecTimeStamp();
39 var ntsmax = new Util.Basic.TimeX.DateTimeX(DateTime.Now.AddDays(1)).ToMiSecTimeStamp();
40 var inbound = from a in Data

42 long.Parse(a.Value?.startAt?[(int)c]??"0") < nts &&
43 long.Parse(a.Value?.endAt?[(int)c] ?? "0") > nts
44 select a.Key;
45 var outbound = from a in Data
